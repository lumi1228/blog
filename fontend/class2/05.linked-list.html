1. 什么是链表

  • 链表是多个元素组成的一种物理结构，类似于数组
  • 元素存储不连续，用next指针连在一起
  • 链表节点的数据结构{value,next?,prev?}
  • javascript 没有链表数据结构，用Object模拟链表；
  • 链表常用操作：修改next，遍历链表
  • 遍历链表，插入、删除

基本使用示例：

链表创建示例
function creatLinkList(arr){
  if(!arr || !arr.length) throw new Error('arr is empty!');
  const length = arr.length
  //从后往前遍历数组
  //获取最后一个值
  let curNode ={
    value:arr[length-1]
  }
  // 若只有一个元素，则直接返回当前节点
  if(length === 1) {
    return curNode
  }
  // 若多个元素，则从后遍历数组，依次这是链表值
  for(let i = length-2 ; i >=0; i--){
    //重新定义节点，并赋值当前节点的next
    curNode ={
      value:arr[i],
      next:curNode
    }
  }
  return curNode
}
链表实现队列示例
// 实现思路：
// * 用单向链表，但是要同时记录head和tail「头、尾」
// * 要从tail入队，从head出队，否则出队时tail不好定位
// * length 要实时记录，不可便利链表获取

class MyQueue{
  constructor(){
    this.head = null
    this.tail = null
    this.length = 0
  }
  add(n){
    let newNode ={
      value : n,
      next:null
    }
    //若链表为空，头尾节点均指向新节点
    if(this.length === 0) {
      this.head = newNode
      this.tail =  newNode
    }else{
      // 入队
      this.tail.next = newNode
      // tail指向新的尾节点
      this.tail=newNode
    }
    this.length ++
    
  }
  delete(){
    let headNode = this.head
    if(!headNode || this.length <=0 ) return null
    
    // 取值
    let value = this.head.value
    // head指向新的头节点
    this.head = this.head.next
    this.length --
  }
  getLength(){
    // length要单独处理，不能遍历获取，否则时间复杂度太高
    return this.length
  }

}

反转单向链表示例
//反转单向链表，并返回反转后的headnode
// 实现思路：
// 反转，将节点next指向前一个节点
// 很容易造成nextNode丢失
// 需要三个指针 prevNode curNode nextNode
function reverseLinkList(listNode) {
  // 定义三个指针，初始值nextNode 指向头节点
  let prevNode  = undefined
  let curNode  = undefined
  let nextNode  = listNode

  while(nextNode){
    // 只有一个元素 删掉next 防止循环引用
    if(curNode && !prevNode) delete curNode.next

    //反转指针
    if(curNode && prevNode) curNode.next = prevNode

    //整体向后移动
    prevNode = curNode 
    curNode = nextNode 
    nextNode = nextNode.next
  }
  // 补充最后一个，当nextNode，curNode未设置next
  curNode.next = prevNode
  return curNode
}

2.链表 VS 数组

  • 都是有序结构
  • 链表：查询慢 O(n), 新增删除快 O(1)[React Fiber使用了链表]
  • 数组：查询快 O(1), 新增删除慢 O(n)
2-1.链表和数组，哪个实现队列更快?
  • 数组是连续存储，push很快、shift很慢
  • 链表是非连续存储，add、delete很快、查找很慢
2-2.扩展

对象是无序结构、map是有序结构 数组是有序结构、set是无序结构

3. 前端与链表

3-1.JS中的原型链

原型链本质是链表
原型链上的节点是各种原型对象，比如 Function.prototype、Object.prototype
原型链通过 proto 属性连接各种原型对象

  • obj -> Object.prototype -> null
  • func -> Function.prototype -> Object.prototype -> null
  • arr -> Array.prototype -> Object.prototype -> null

    while (A && B){
        if (A.__proto__ === B.prototype) {
            return true
